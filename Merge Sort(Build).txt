Merge Sort(Build):
#include<stdio.h>
#include<stdlib.h>
void merge(int a[],int l,int r,int mid,int s)
{
	int left[l],right[r],i;
	for(i=0;i<mid;i++)
	{
		left[i]=a[i];
	}
	for(i=mid;i<s;i++)
	{
		right[i-mid]=a[i];
	}
	int j,k=0;
	i=0;
	while(i<l && j<r)
	{
		if(left[i]<=right[i])
		{
			a[k]=left[i];
			i++;
		}
		else
		{
			a[k]=right[j];
			j++;
		}
		k++;
	}
	while(i<l)
	{
		a[k]=left[i];
		i++;
		k++;
	}
	while(j<r)
	{
		a[k]=right[j];
		j++;
		k++;
	}
}
void mergesort(int a[],int s)
{
	int mid,l,r,least,max;
	least=0;
	max=s-1;
	mid=least + (max - least) / 2;
	l=mid;
	r=s-mid;
	mergesort(a,l);
	mergesort(a,r);
	merge(a,l,r,mid,s);
}
int main()
{
	int a[8];
	int s=sizeof(a)/sizeof(a[0]);
	int i;
	printf("Enter 8 integers : ");
	for(i=0;i<s;i++)
	{
		scanf("%d",&a[i]);
	}
	mergesort(a,s);
	for(i=0;i<s;i++)
	{
		printf("%d",a[i]);
	}
}
